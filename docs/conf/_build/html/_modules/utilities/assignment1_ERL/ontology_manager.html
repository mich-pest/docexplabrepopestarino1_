    """
    This is a class used to simplify the task related to load, query and replace commands sent to the ARMOR service, needed to integrate the ontology manipulation in the software framework.
    
    Attributes:
        _client_name (str): Module-level variable used to keep track of single client name, needed to distinguish among several possible clients in complex scenarios.
        _reference_name (str): Module-level variable used to keep track of single reference name for the ontology, needed to distinguish among several possible ontologies in complex scenarios.
        _timeout (int): Maximum time spent waiting for the service before returning an error.
        _service_name (str): Identifier for the service, needed to instanciate the client and perform calls.
        _client (ServiceProxy): Proxy item used to call the ARMOR service.
    """
    def __init__(self, name, ont_name, service_name=anm.ARMOR_SERVICE, timeout=5):
        self._client_name = name
        self._reference_name = ont_name
        self._timeout = timeout
        self._service_name = service_name
        self._client = rospy.ServiceProxy(self._service_name, ArmorDirective)
    

[docs]    def replace_objectprop_ind(self, objectprop_name, ind_name, new_value, old_value):
        """
        Function to replace the value of an object property belonging to an individual. 
        If the individual or the property instance to be replaced do not exist, they will be created.
    
        Args:
            objectprop_name (str): name of the object property to replace.
            ind_name (str): individual whose property has to be replaced.
            new_value (str): new value of the object property.
            old_value (str): current value of the object property to be replaced.
    
        Raises:
            rospy.ServiceException: if call to ARMOR fails.

        Returns:
            bool: True if ontology is consistent, otherwise False.
    
        Note:
            Again, *old_value* is necessary since more than one object property instance may be applied to an individual. 
            Only the specified instance will be replaced.
        """
        try:
            res = self.call('REPLACE', 'OBJECTPROP', 'IND', [objectprop_name, ind_name, new_value, old_value])
    
        except rospy.ServiceException:
            log_msg = "Service call failed upon adding property {0} to individual {1}.".format(objectprop_name, ind_name)
            rospy.logerr(anm.tag_log(log_msg, self._client_name))

        if res.success:
            return res.is_consistent
        else:
            raise rospy.ServiceException(res.error_description, res.exit_code)

        """
        Function to replace the value of a data property belonging to an individual. 
        If the individual or the property instance to be replaced do not exist, they will be created.
    
        Args:
            dataprop_name (str): name of the data property to assign.
            ind_name (str): individual to assign the data property value.
            value_type (str): type of the value to assign (INTEGER, INT, FLOAT, LONG, DOUBLE, STRING, BOOLEAN, BOOL).
            new_value (str): new value of the data property.
            old_value (str): current value of the data property to be replaced.
    
        Raises:
            rospy.ServiceException: if call to ARMOR fails.
            
        Returns:
            bool: True if ontology is consistent, otherwise False.
    
        Note:
            *old_value* is necessary since more than one object property instance may be applied to an individual. 
            Only the specified instance will be replaced.
    
        Note:
            Consistency can break even if you send a proper request but the ontology is expecting a different value type.
        """
        try:
            res = self.call('REPLACE', 'DATAPROP', 'IND', 
                                   [dataprop_name, ind_name, value_type, new_value, old_value])
    
        except rospy.ServiceException:
            log_msg = "Service call failed upon adding property {0} to individual {1}.".format(dataprop_name, ind_name)
            rospy.logerr(anm.tag_log(log_msg, self._client_name))

        if res.success:
            return res.is_consistent
        else:
            raise rospy.ServiceException(res.error_description, res.exit_code)

        """
        Function to query the value of an object property associated with an individual.
    
        Args:
            objectprop_name (str): object property that is queried.
            ind_name (str): individual whose property is queried.

        Raises:
            rospy.ServiceException: if call to ARMOR fails.

        Returns:
            str: parsed result returned by the service.
        """
        try:
            res = self.call('QUERY', 'OBJECTPROP', 'IND', [objectprop_name, ind_name])
    
        except rospy.ServiceException:
            log_msg = "Service call failed upon querying property {0} to individual {1}.".format(objectprop_name, ind_name)
            rospy.logerr(anm.tag_log(log_msg, self._client_name))
    
        if res.success:
            parsed_res = [item.split('#')[1].split('>')[0] for item in res.queried_objects]
            return parsed_res
        else:
            raise rospy.ServiceException(res.error_description, res.exit_code)

[docs]    def sync_reasoner(self):
        """
        Function to sync ontology reference reasoner.
    
        Raises:
            rospy.ServiceException: if call to ARMOR fails.

        Returns:
            bool: True if ontology is consistent, otherwise False.
        """
        try:
            res = self.call('REASON', '', '', [])
            rospy.loginfo(
                "Reasoner synced on {0} applied, consistency: {1}".format(self._reference_name, res.is_consistent))

        except rospy.ServiceException:
            log_msg = "Armor service internal error. Buffered reasoner could not be sync."
            rospy.logerr(anm.tag_log(log_msg, self._client_name))

        if res.success:
            return res.is_consistent
        else:
            raise rospy.ServiceException(res.error_description, res.exit_code)



[docs]    def save(self, filepath, inference=False):
        """
        Function to save the ontology to a given path.
    
        Args:
            filepath (str): location of the saved ontology.
    
        Raises:
            rospy.ServiceException: if save operation not successful.
        """
        try:
            if inference:
                res = self.call('SAVE', 'INFERENCE', '', [filepath])            
            else:
                res = self.call('SAVE', '', '', [filepath])
            rospy.loginfo(
                "Reference {0} saved to {1}".format(self._reference_name, filepath))

        except rospy.ServiceException as e:
            log_msg = "Armor service error. Cannot save reference {0} to {1}.".format(self._reference_name, filepath)
            rospy.logerr(anm.tag_log(log_msg, self._client_name))



[docs]    def load_ontology_from_file(self, owl_file_path, iri,
                           buffered_manipulation=False, reasoner="PELLET", buffered_reasoner=True, mounted=False):
        """
        Function to load an ontology into ARMOR from an .owl file.
    
        Args:
            owl_file_path (str): location of the ontology in the system.
            iri (str): iri path given by the user.
            buffered_manipulation (bool): set buffered manipulations, default False.
            reasoner (str): set which reasoner to use (PELLET, HERMIT, SNOROCKET, FACT), default PELLET.
            buffered_reasoner (bool): set if reasoner should be buffered, default False.
            mounted (bool): set if the client should be mounted on the reference, default False.
    
        Raises:
            rospy.ServiceException: if call to ARMOR fails.

        Returns:
            bool: True if the call is successful, otherwise False.
        """
        try:
            if mounted:
                res = self.call('LOAD', 'FILE', 'MOUNTED',
                                        [owl_file_path, iri, str(buffered_manipulation), reasoner, str(buffered_reasoner)])
            else:
                res = self.call('LOAD', 'FILE', '',
                                        [owl_file_path, iri, str(buffered_manipulation), reasoner, str(buffered_reasoner)])
            return res.success
        
        except rospy.ServiceException:
            log_msg = "Service call failed upon reference {0} from {1}".format(self._reference_name, self._client_name)
            rospy.logerr(anm.tag_log(log_msg, self._client_name))

            rospy.logerr(anm.tag_log(log_msg, self._client_name))

        if res.success:
            parsed_res = [item.split('#')[1].split('>')[0] for item in res.queried_objects]
            #print("CLASSES FOR " + str(ind_name) + " are: " + str(parsed_res))
            return parsed_res
        else:
            raise rospy.ServiceException(res.error_description, res.exit_code)

[docs]    def filter_ind_by_class(self, ind_names, class_name):
        """
        Function to filter individuals from a list, depending on the classes they belong to.
        If the individual belongs to `class_name` it will appear in the first list returned by the function, otherwise in the second one.

        Args:
            ind_names (list): list of strings, each of them referred to an individual of the ontology.
            class_name (str): class used to filter the individuals.
        
        Returns:
            list: list of individuals belonging to the class `class_name`.
            list: list of individuals not belonging to the class `class_name`.
        """

        inds_filtered_in = []
        inds_filtered_out = []
        for r in ind_names:
            if class_name in self.get_ind_classes(r): # if ind is of type class_name
                inds_filtered_in.append(r)
            else:
                inds_filtered_out.append(r) 
        return inds_filtered_in, inds_filtered_out


    def _prepare_request(self, command, first_spec, second_spec, args_list):
        """
        Function to filling in the fields of a request to the ARMOR service. 

        Args:
            command (str): string that identifies the command of the request.
            first_spec (str): string that identifies the primary specification of the request.
            second_spec (str): string that identifies the secondary specification of the request.
            args_list (list): list that contains all needed arguments to send a proper request.

        Returns:
            ArmorDirectiveReq: ARMOR request ready to be sent.
        """
        req = _ArmorDirectiveReq.ArmorDirectiveReq()
        req.client_name = self._client_name
        req.reference_name = self._reference_name
        req.command = command
        req.primary_command_spec = first_spec
        req.secondary_command_spec = second_spec
        req.args = args_list
        return req
    

[docs]    def call(self, command, first_spec, second_spec, args_list):
        """
        Function to perform the actual call to the service, if it is already available.

        Args:
            command (str): string that identifies the command of the request.
            first_spec (str): string that identifies the primary specification of the request.
            second_spec (str): string that identifies the secondary specification of the request.
            args_list (list): list that contains all needed arguments to send a proper request.

        Returns:
            ArmorDirectiveRes: ARMOR response to the queried operation. 
        """
        req = self._prepare_request(command, first_spec, second_spec, args_list)
        rospy.wait_for_service(self._service_name, self._timeout)
        res = self._client(req).armor_response
        return res
