    """
    Class to create and easily deal with SimpleActionClient entities.

    Attributes:
        _server_name (str): Identifier for the server, needed to instanciate the client. 
        _client (SimpleActionClient): Client item used to send goals to the action server.
        _external_done_cb (function): Callback executed any time goal execution is terminated.
        _external_feedback_cb (function): Callback executed any time new feedback is provided.
        _mutex (lock): mutex defined for concurrent execution management. 
        _is_running (bool): Boolean flag to discriminate if the client sent a goal (the server is running it) or not.
        _is_done (bool): Boolean flag to discriminate if the goal execution is finished or not. 
        _results (resultType (its type will depend on the specific action definition)): Actually results of server computation.
    """
    def __init__(self, server_name, action_type, done_callback = None, feedback_callback = None, mutex = None):
        self.reset_states()
        self._server_name = server_name

        if mutex is None:
            self._mutex = Lock()
        else: 
            self._mutex = mutex

        self._client = SimpleActionClient(server_name, action_type)
        self._external_done_cb = done_callback
        self._external_feedback_cb = feedback_callback
        self._client.wait_for_server()


[docs]    def send_goal(self, goal):
        """
        Asynchronously executed function to send a new goal to the ActionServer that also modifies internal flag values.

        Args:
            goal (goalType (its type will depend on the specific action definition)): goal to be sent.
        """
        if not self._is_running:
            self._client.send_goal(goal, done_cb = self._done_callback, feedback_cb = self._feedback_callback)
            self._is_running = True
            self._is_done = False
            self._results = None

        """
        Function to reset internal flag values needed for robust execution.
        """
        self._is_running = False

        self._results = None



[docs]    def cancel_goals(self):
        """
        Function to cancel the running goal.
        """
        if self._is_running:
            self._client.cancel_all_goals()
            self.reset_states()


    def _feedback_callback(self, feedback):
        """
        Function executed while goal is running, giving some feedback information.

        Args:
            feedback (feedbackType (its type will depend on the specific action definition)): feedback message sent by the server.
        """
        self._mutex.acquire()
        try:
            if self._external_feedback_cb is not None:
                print('`' + self._server_name + '` action server provides feedback' + str(feedback))
                self._external_feedback_cb(feedback)
        finally:
            self._mutex.release()

    def _done_callback(self, status, results):
        """
        Function executed when goal is completed, giving some status information.

        Args:
            status (SimpleClientGoalState): status of goal execution (PENDING, ACTIVE, REJECTED, SUCCEEDED, ABORTED, PREEMPTING, PREEMPTED, RECALLING, RECALLED, LOST).
            results (resultType (its type will depend on the specific action definition)): actual result of the execution.
        """
        self._mutex.acquire()
        try:
            self._is_running = False
            self._is_done = True
            self._results = results
            if self._external_done_cb is not None:
                print('`' + self._server_name + '` done with state `' + self._client.get_state_txt() + '` and result:' + str(results))
                self._external_done_cb(status, results)
        finally:
            self._mutex.release()

    """
    Class to create and easily deal with shared variables updated by ROS topics content, instances of the ActionClientHelper class as well as debug features provided by the software framework.

    Attributes:
        _track_position (bool): Flag needed to enable or disable position tracking of the robot for debug purposes.
        _log_file (TextIOWrapper): Reference to the log file in which robot position is reported, if `_track_position` feature is enabled. 
        mutex (lock): Mutex defined for exclusive access to _battery_low variable and for ActionClientHelper callback execution.
        _battery_low (bool): Attribute to keep track of current battery status.
        _robot_name (str): Label for robot individual in the ontology, retrieved at the beginning and kept as a constant while executing.
        planner_client (ActionClientHelper): Client needed to generate the plan through the `motion/planner` action server to reach a new target room given its name.
        controller_client (ActionClientHelper): Client needed to generate commands for the controller through the `motion/controller` action server for executing the plan.
        set_pose_service (ServiceProxy): Proxy item needed to randomly set robot position at the beginning of the execution (to have random planner and controller working).
        _save_ontology (bool): Flag value to dictate if the modified ontology has to saved.
        _ontology_output_path (str): Path to which the ontology has to be saved.
        _urgency_threshold (int): Maximum time for which a room is not visited before consider it urgent.
        _robust_mode (bool): Flag to discriminate the way in which the urgency is checked.

    Note:
        Planner and controller components, being just placeholders for future development, execute randomly only simulating a real plan generation and actuators motion. 
    """
    def __init__(self):
        """
        Raises:
            rospy.ServiceException: in case of failed call to the service.
        """
        self._track_position, self._save_ontology = self.get_params([anm.TRACK_POSITION_FLAG, anm.SAVE_ONTOLOGY])
        if self._save_ontology:
            self._ontology_output_path = self.get_params([anm.SAVE_ONTOLOGY_PATH])[0]
        self._log_file = ''
        self.mutex = Lock()
        self._battery_low = False
        self._robot_name = ''
        self._urgency_threshold = 0
        self._robust_mode = False

        # Subscriber for battery topic 
        rospy.Subscriber(anm.TOPIC_BATTERY_LOW, Bool, self._battery_callback)
        
        # Clients for planner and controller
        self.planner_client = ActionClientHelper(anm.ACTION_PLANNER, PlanAction, mutex = self.mutex)
        self.controller_client = ActionClientHelper(anm.ACTION_CONTROLLER, ControlAction, mutex = self.mutex)
        
        # Setting initial robot's position
        rospy.wait_for_service(anm.SERVER_SET_POSE)
        try:
            self.set_pose_service = rospy.ServiceProxy(anm.SERVER_SET_POSE, SetPose)
        except rospy.ServiceException as e:
            print("Service call failed: %s"%e)

        self.set_robot_pose()

        if self._track_position:
            self.init_log_file()
        
            
    def _battery_callback(self, msg):
        """
        Asynchronously executed function to update battery status.

        Args:
            msg (bool): value indicating the battery status: if true the battery is low, otherwise the battery does not force the robot to stop and recharge.
        """
        self.mutex.acquire()
        try:
            self._battery_low = msg.data
            if self._battery_low:
                print('Robot with low battery')
            else:
                print('Robot battery fully charged')
        finally: 
            self.mutex.release()

[docs]    def get_params(self, params):
        """
        Function to get the expected parameters from the ROS parameters server.

        Args:
            params (list): list of keys associated with the parameters to be retrieved.

        Raises:
            rospy.ROSInitException: if at least one expected parameter is not provided.

        Returns:
            list: list of all requested parameters associated with the given keys.
        """
        resp = []
        for p in params:
            if rospy.has_param(p):
                resp.append(rospy.get_param(p))
        if len(resp) != len(params):
            err_msg = "ERROR: not enough parameters provided"
            raise rospy.ROSInitException(err_msg)
        else:
            return resp
